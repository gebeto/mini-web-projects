If we feel like it, we can also use a computed property for the description.

enum Season {
  case spring, summer, fall, winter
  var description: String {
    switch self {
    case .summer:
      return "hot season"
    case .winter:
      return "cold season"
    case .spring, .fall:
      return "transitional season"
    }
  }
}
var season = Season. spring summer fall winter
print(season.description)
transitional season hot season transitional season cold season
