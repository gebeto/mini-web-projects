For the gold challenge, add the colors as an array to the menu view controller and keep track of the current color with an index instance variable. Finally, schedule a timer and have it trigger every second.

class MenuViewController: UIViewController {

var index = 0
let colors = [UIColor.cyan, UIColor.brown, UIColor.purple]

override func viewDidLoad() {
super.viewDidLoad()
Timer.scheduledTimer(timeInterval: 1, target: self, selector: #selector(changeColor), userInfo: nil, repeats: true)
}

func changeColor() {
self.index += 1
if self.index &gt; 2 {
self.index = 0
}
self.view.backgroundColor = self.colors[self.index]
}

}
