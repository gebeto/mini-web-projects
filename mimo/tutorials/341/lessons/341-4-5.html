Now let's create a way to check if the user's guess was correct!

Add a guess() function that takes an Int-type guess as argument, compares it with the random number, and returns a boolean.

import Foundation

class RandomizlyModel {
var number = 0

init() {
self.newNumber()
}

func newNumber() {
self.number = Int(arc4random_uniform(6)+1)
}

func guess(_ guess: Int) -&gt; Bool {
return self.number == guess
}

}
