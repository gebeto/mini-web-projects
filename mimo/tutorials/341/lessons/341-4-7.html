Call the model's guess() function from the view controller's guess() function with self.model.guess(), passing the number variable to the function.

@IBAction func guess(_ sender: Any) {
if let input = self.inputField.text, let number = Int(input) {
self.model.guess(number)
print("Input: \(number)")
} else {
print("No input")
self.showAlert()
}
}
