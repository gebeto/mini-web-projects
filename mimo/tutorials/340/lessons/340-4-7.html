For the gold challenge, we can bundle the UIColor-type colors in a colors array and keep track of the current color with an index instance variable.

class ViewController: UIViewController {

@IBOutlet weak var randomNumberLabel: UILabel!

var number = 10
var index = 0
let colors = [UIColor.red, UIColor.green, UIColor.blue]

override func viewDidLoad() {
super.viewDidLoad()
self.newRandomNumber(self)
}

@IBAction func newRandomNumber(_ sender: Any) {
self.number = Int(arc4random_uniform(6))
self.randomNumberLabel.text = "\(number)"

self.index += 1
if self.index > 2 {
self.index = 0
}
self.view.backgroundColor = self.colors[self.index]

}

}

Then we increase index by 1, making sure that it's always between 0 and 2, pick a color from colors and display it.
