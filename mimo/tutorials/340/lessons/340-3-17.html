Now copy-paste the code from viewDidLoad() to newRandomNumber(), so that the app also does its random number magic when we push the button.

class ViewController: UIViewController {

@IBOutlet weak var randomNumberLabel: UILabel!

var number = 10

override func viewDidLoad() {
super.viewDidLoad()
}

@IBAction func newRandomNumber(_ sender: Any) {
self.number = Int(arc4random_uniform(6))
self.randomNumberLabel.text = "\(number)"
}
}

When you run the app, a different random number should appear every time you hit Regenerate.
