Let's create a numberinstance variable, place it somewhere between the class declaration and viewDidLoad(), and give it a value of 10!

import UIKit

class ViewController: UIViewController {

@IBOutlet weak var randomNumberLabel: UILabel!
var number = 10

override func viewDidLoad() {
super.viewDidLoad()
let number = 10
self.randomNumberLabel.text = "\(number)"
}

}
