For the silver challenge, simply add commas and terminator: " " so that print() adds a single space at the end of the line instead of a line break.

var number = 10
number = 11

if number == 10 {
print("Guessed correctly!", terminator:" ")
} else if number == 11 {
print("Second try!", terminator:" ")
} else {
print("What's your number?", terminator:" ")
}

print("The number is \(number).")
