"Agent Smith" is about to defeat the plans of the only remaining villain in town. For this special mission, however, he'll need a special plan.

Create a namespace and a class and use args from Main() for the villain's name and location. Pass them to the constructor of a Plan class, which also has an Execute() method.

using System;
namespace Mission
{
 class Program
 {
  public static void Main(string[] args)
  {
   string villain = args[0];
   string location = args[1];
   Plan plan = new Plan(villain, location);
   plan.Execute();
  }
 }
}
using System;
namespace Mission
{
 class Plan
 {
  private string location;
  private string villain; 
  public Plan(string location, string villain)
  {
   this.location = location;
   this.villain = villain;
  }   
  public void Execute()
  {
   Console.WriteLine("Traveling to " + location);
   /* Qualified information */
   Console.WriteLine("Defeated " + villain + "'s plans");
  }
 }
}
