Let's continue with the project! What if we wanted to translate a different text into British English? Wouldn't it be great if we could reuse the code we've written?

Make the translator script a translate() function that accepts a text parameter and returns the translated text. Invoke the function and display the results.

function translate(text) {
 var words = text.split(" ");
 for (var i = 0; i < words.length; i++) {
  if (words[i].endsWith("ize")) {
   words[i] = words[i].replace("ize", "ise");
  } else if (words[i].endsWith("or")) {
   words[i] = words[i].replace("or", "our");
  }
 }
 text = words.join(" ");
 return text;
}

var aeText = "Repetition, simple sentences, and supporting images familiarize the beginning reader with the color blue.";
var beText = translate(aeText);
console.log(beText);
