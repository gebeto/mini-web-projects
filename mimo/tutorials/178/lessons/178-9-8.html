It's time for the ultimate project, an object for the morning routine.

Create a constructor function that uses a date parameter. Then move the rest of the script into a wakeUp() method and change it to use the date property.

function Morning(date) {
 this.date = date;
 this.wakeUp() = function() {
  var days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
  var dayNumber = date.getDay();
  switch (dayNumber) {
  case 0: // Sunday
  case 6: // Saturday
   this.date.setHours(10);
   break;
  default:
   this.date.setHours(7);
  }
  date.setMinutes(0);
  date.setSeconds(0);

  var day = days[dayNumber];
  console.log("It's " + day + ", " + date.toTimeString() + ". Time to get up.");
}
