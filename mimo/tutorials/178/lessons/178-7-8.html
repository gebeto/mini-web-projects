For this project, let's take another look at the translator script! What if a word in the text contains "ize" or "or" but doesn't end with it?

Split the text into words that you store in a words variable. Iterate over them and make sure that they end with the "ize" or "or" before your replace anything.

var text = "Repetition, simple sentences, and supporting images familiarize the beginning reader with the color blue.";

var words = text.split(" ");
for (var i = 0; i < words.length; i++) {
 if (words[i].endsWith("ize")) {
  words[i] = words[i].replace("ize", "ise");
 } else if (words[i].endsWith("or")) {
  words[i] = words[i].replace("or", "our");
 }
}
text = words.join(" ");
console.log(text);
