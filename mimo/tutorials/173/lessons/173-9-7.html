For the Cookie class let's create a get_flavor() method to display what the cookie tastes like!

class Cookie: 
  def __init__ (self, ingredients): 
    self.dough = ingredients
  def get_flavor(self):
    return self.dough[0]

cookie = Cookie(["honey", "nuts"])
flavor = cookie. get_flavor() get_flavor
print("Tastes like " + flavor)
Tastes like honey
